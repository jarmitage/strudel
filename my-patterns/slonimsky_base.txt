// Beyond Slonimsky: Melodic Patterns for Guitar, Rich Cochrane, p.3
// Slonimsky's first recipe works as follows. 
// First take a certain number of octaves. 
// Then divide them into equal parts: 
// the first note of each part goes into the 'base'. 
// That's all there is to it. 
// Slonimsky is interested in all the possible ways to do this.

let octs = "[12 24 36 48 60]/20"

saw.range(0,octs).floor().segment(octs)
  .add((octs).outside(20, rev))
  .struct("x*<3 4 6 12>")
  .note().pianoroll()
